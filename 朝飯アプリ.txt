<script>
(() => {
  const KEY = "breakfast_log_v1";

  const $ = (id) => document.getElementById(id);

  function pad(n){ return String(n).padStart(2,"0"); }

  // ローカル日付キー（例: 2026-02-18）を作る（TZは端末のローカル）
  function toDayKey(d){
    return `${d.getFullYear()}-${pad(d.getMonth()+1)}-${pad(d.getDate())}`;
  }

  function fmtDateJP(dayKey){
    const [y,m,da] = dayKey.split("-").map(Number);
    return `${y}年${m}月${da}日`;
  }

  function fmtTimeJP(d){
    return `${pad(d.getHours())}:${pad(d.getMinutes())}:${pad(d.getSeconds())}`;
  }

  function load(){
    try {
      const raw = localStorage.getItem(KEY);
      if(!raw) return { events: [] };
      const data = JSON.parse(raw);
      if(!data || !Array.isArray(data.events)) return { events: [] };
      return data;
    } catch {
      return { events: [] };
    }
  }

  function save(data){
    localStorage.setItem(KEY, JSON.stringify(data));
  }

  // events: [{ts:number, dayKey:string}]
  function addEvent(){
    const data = load();
    const now = new Date();
    const dayKey = toDayKey(now);

    // 1日に複数押しても「最新1件」だけ残す（重複ウザいやろ）
    const filtered = data.events.filter(e => e.dayKey !== dayKey);
    filtered.push({ ts: now.getTime(), dayKey });
    filtered.sort((a,b) => a.ts - b.ts);

    save({ events: filtered });
  }

  function removeToday(){
    const data = load();
    const todayKey = toDayKey(new Date());
    const filtered = data.events.filter(e => e.dayKey !== todayKey);
    save({ events: filtered });
  }

  function resetAll(){
    localStorage.removeItem(KEY);
  }

  function uniqueDayKeys(events){
    const set = new Set(events.map(e => e.dayKey));
    return Array.from(set).sort();
  }

  function dayRange(startKey, endKey){
    // startKey〜endKey（両端含む）の配列
    const [sy,sm,sd] = startKey.split("-").map(Number);
    const [ey,em,ed] = endKey.split("-").map(Number);
    const start = new Date(sy, sm-1, sd);
    const end = new Date(ey, em-1, ed);

    const out = [];
    const cur = new Date(start);
    while(cur <= end){
      out.push(toDayKey(cur));
      cur.setDate(cur.getDate() + 1);
    }
    return out;
  }

  function calcStats(events){
    const days = uniqueDayKeys(events);
    if(days.length === 0){
      return {
        start: null,
        totalDays: 0,
        missed: 0,
        streakNow: 0,
        streakBest: 0,
        daysSet: new Set()
      };
    }

    const startKey = days[0];
    const todayKey = toDayKey(new Date());
    const allDays = dayRange(startKey, todayKey);

    const daySet = new Set(days);
    const totalDays = daySet.size;

    let missed = 0;
    for(const k of allDays){
      if(!daySet.has(k)) missed++;
    }

    // 最高ストリーク：allDaysを走査して連続数の最大
    let best = 0, cur = 0;
    for(const k of allDays){
      if(daySet.has(k)){
        cur++;
        if(cur > best) best = cur;
      } else {
        cur = 0;
      }
    }

    // 今のストリーク：今日から遡って連続
    let streakNow = 0;
    // allDaysの末尾（today）から逆
    for(let i = allDays.length - 1; i >= 0; i--){
      if(daySet.has(allDays[i])) streakNow++;
      else break;
    }

    return { start: startKey, totalDays, missed, streakNow, streakBest: best, daysSet: daySet };
  }

  function render(){
    const data = load();
    const events = data.events.slice().sort((a,b) => b.ts - a.ts); // 最新が上

    // Now表示
    const now = new Date();
    $("now").textContent = `いま: ${now.getFullYear()}-${pad(now.getMonth()+1)}-${pad(now.getDate())} ${fmtTimeJP(now)}`;

    $("tzBadge").textContent = Intl.DateTimeFormat().resolvedOptions().timeZone || "Local";

    // Stats
    const stats = calcStats(data.events);
    $("kpiStart").textContent = stats.start ? fmtDateJP(stats.start) : "-";
    $("kpiTotalDays").textContent = String(stats.totalDays);
    $("kpiMissed").textContent = String(stats.missed);
    $("kpiStreak").textContent = `${stats.streakNow} / ${stats.streakBest}`;

    $("startInfo").textContent = stats.start
      ? `開始: ${fmtDateJP(stats.start)} 〜 今日まで`
      : "まだ1回も記録なし。まず押し。";

    // 今日取り消しボタンは、今日記録がある時だけ有効
    const todayKey = toDayKey(new Date());
    $("btnUndo").disabled = !stats.daysSet.has(todayKey);

    // Log table
    const tbody = $("logTable");
    tbody.innerHTML = "";
    events.forEach((e, idx) => {
      const d = new Date(e.ts);
      const tr = document.createElement("tr");
      tr.innerHTML = `
        <td>${events.length - idx}</td>
        <td>${fmtDateJP(e.dayKey)}</td>
        <td>${fmtTimeJP(d)}</td>
      `;
      tbody.appendChild(tr);
    });
    if(events.length === 0){
      const tr = document.createElement("tr");
      tr.innerHTML = `<td colspan="3" class="muted">まだ履歴なし。右上のボタン押したら増えるで。</td>`;
      tbody.appendChild(tr);
    }

    // Calendar: 直近28日
    const cal = $("calendar");
    cal.innerHTML = "";

    const daysBack = 28;
    const end = new Date();
    const start = new Date();
    start.setDate(end.getDate() - (daysBack - 1));

    const range = [];
    const cur = new Date(start);
    while(cur <= end){
      range.push(new Date(cur));
      cur.setDate(cur.getDate() + 1);
    }

    range.forEach(d => {
      const key = toDayKey(d);
      const box = document.createElement("div");
      const has = stats.daysSet.has(key);
      box.className = `day ${has ? "ok" : "ng"}`;
      box.innerHTML = `
        <div class="d">${pad(d.getMonth()+1)}/${pad(d.getDate())}</div>
        <div class="m">${has ? "✅" : "❌"}</div>
      `;
      cal.appendChild(box);
    });
  }

  // Events
  $("btnEat").addEventListener("click", () => {
    addEvent();
    render();
  });

  $("btnUndo").addEventListener("click", () => {
    removeToday();
    render();
  });

  $("btnReset").addEventListener("click", () => {
    const ok = confirm("全データ削除するで？ほんまにええん？（戻せへん）");
    if(ok){
      resetAll();
      render();
    }
  });

  // 初期描画 & 時計更新
  render();
  setInterval(render, 1000);
})();
</script>
